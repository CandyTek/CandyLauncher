  加载时机策略

  enum class LoadTiming {
      OnStartup,          // 程序启动时
      OnDemand,           // 按需加载
      OnFirstUse,         // 首次使用时
      OnSchedule,         // 计划时间
      OnTrigger          // 特定触发事件
  };

  struct LoadingStrategy {
      LoadTiming timing;
      std::string condition;      // 加载条件（如文件路径、时间等）
      int priority = 0;           // 加载优先级
      bool background = false;    // 是否后台加载
      std::chrono::seconds delay{0}; // 延迟加载时间
  };

  class PluginLoadingScheduler {
  private:
      std::map<std::string, LoadingStrategy> m_strategies;
      std::priority_queue<ScheduledLoad> m_loadQueue;
      std::thread m_schedulerThread;

      struct ScheduledLoad {
          std::string pluginName;
          std::chrono::steady_clock::time_point when;
          int priority;
          bool operator<(const ScheduledLoad& other) const {
              if (when == other.when) return priority < other.priority;
              return when > other.when; // 反向，早的优先
          }
      };

  public:
      void SetLoadingStrategy(const std::string& plugin, const LoadingStrategy& strategy);
      void ScheduleLoad(const std::string& plugin);
      void TriggerLoad(const std::string& plugin, const std::string& trigger);
      void StartScheduler();
      void StopScheduler();
  };

  回答用户关于 allActions 的疑问

  关于你提到的"别人已经是dll了，它也有权限有线程可以更改我的 allActions 列表？"这个问题：

  安全考虑：
  1. 不建议直接暴露 allActions：插件不应该直接访问你的内部数据结构
  2. 使用代理模式：通过 SecureDataProxy 提供受控访问
  3. 权限分级：不同插件有不同的访问权限
  4. 操作审计：记录所有数据访问操作
  5. 沙箱隔离：限制插件的系统访问能力

  推荐方案：
  - 插件通过 IPluginHost 接口请求操作
  - 主程序验证权限后执行操作
  - 所有数据修改都通过主程序的API进行
  - 实现数据同步机制处理并发访问

  这样既保证了功能灵活性，又维护了系统安全性。

  #pragma once

  void FileSearchPlugin::LoadConfiguration() {
      // 加载搜索路径
      std::string searchPaths = GetConfig("search_paths", "C:\\Users;D:\\");
      ParseSearchPaths(searchPaths);

      // 加载文件扩展名过滤
      std::string extensions = GetConfig("file_extensions", ".txt,.doc,.docx,.pdf,.jpg,.png");
      ParseFileExtensions(extensions);

      // 加载其他配置
      m_searchSubdirectories = GetConfigBool("search_subdirectories", true);
      m_maxResults = GetConfigInt("max_results", 20);
      m_minQueryLength = GetConfigInt("min_query_length", 2);
  }

  void FileSearchPlugin::OnInputReceived(const std::string& input) {
      // 检查是否为文件搜索查询
      if (input.length() >= m_minQueryLength) {
          // 检查是否包含文件搜索关键词或文件扩展名
          bool isFileQuery = false;

          // 检查文件扩展名
          for (const auto& ext : m_fileExtensions) {
              if (PluginHelper::StringContains(input, ext)) {
                  isFileQuery = true;
                  break;
              }
          }

          // 检查搜索关键词
          if (!isFileQuery) {
              std::vector<std::string> fileKeywords = {"文件", "file", "搜索", "search", "查找", "find"};
              for (const auto& keyword : fileKeywords) {
                  if (PluginHelper::StringContains(input, keyword)) {
                      isFileQuery = true;
                      break;
                  }
              }
          }

          if (isFileQuery) {
              PerformSearch(input);
          }
      }
  }

  void FileSearchPlugin::PerformSearch(const std::string& query) {
      // 避免重复搜索
      if (m_searchInProgress || query == m_lastQuery) {
          return;
      }

      m_lastQuery = query;
      m_searchInProgress = true;

      // 在新线程中执行搜索
      if (m_searchThread.joinable()) {
          m_searchThread.join();
      }

      m_searchThread = std::thread([this, query]() {
          std::vector<SearchResult> results;

          try {
              for (const auto& searchPath : m_searchPaths) {
                  if (!m_searchInProgress) break;

                  if (PluginHelper::DirectoryExists(searchPath)) {
                      SearchInDirectory(searchPath, query, results);
                  }
              }

              // 按相关性排序
              std::sort(results.begin(), results.end(),
                       [](const SearchResult& a, const SearchResult& b) {
                           return a.relevanceScore > b.relevanceScore;
                       });

              // 限制结果数量
              if (results.size() > m_maxResults) {
                  results.resize(m_maxResults);
              }

              // 更新搜索结果
              UpdateSearchResults(results);

              // 缓存结果
              m_searchCache[query] = results;
              m_lastSearchTime = std::chrono::steady_clock::now();

          } catch (const std::exception& ex) {
              PLUGIN_LOG_ERROR(GetName(), "搜索过程中发生错误: " + std::string(ex.what()));
          }

          m_searchInProgress = false;
      });
  }

  void FileSearchPlugin::SearchInDirectory(const std::string& directory, const std::string& pattern,
                                          std::vector<SearchResult>& results) {
      try {
          std::filesystem::path dirPath(directory);

          auto searchIterator = m_searchSubdirectories ?
              std::filesystem::recursive_directory_iterator(dirPath) :
              std::filesystem::directory_iterator(dirPath);

          for (const auto& entry : searchIterator) {
              if (!m_searchInProgress) break;

              if (entry.is_regular_file()) {
                  std::string fileName = entry.path().filename().string();
                  std::string fileExt = entry.path().extension().string();

                  // 检查文件扩展名过滤
                  if (!m_fileExtensions.empty()) {
                      bool matchExtension = false;
                      for (const auto& ext : m_fileExtensions) {
                          if (PluginHelper::StringEndsWith(fileName, ext, true)) {
                              matchExtension = true;
                              break;
                          }
                      }
                      if (!matchExtension) continue;
                  }

                  // 检查文件名匹配
                  if (PluginHelper::StringContains(fileName, pattern, true)) {
                      SearchResult result;
                      result.filePath = entry.path().string();
                      result.fileName = fileName;
                      result.directory = entry.path().parent_path().string();
                      result.fileSize = entry.file_size();
                      result.lastModified = entry.last_write_time();
                      result.relevanceScore = CalculateRelevanceScore(fileName, pattern);

                      results.push_back(result);
                  }
              }
          }
      } catch (const std::exception& ex) {
          PLUGIN_LOG_WARNING(GetName(), "搜索目录失败: " + directory + ", 错误: " + ex.what());
      }
  }

  void FileSearchPlugin::UpdateSearchResults(const std::vector<SearchResult>& results) {
      // 清除之前的搜索结果
      for (int i = 0; i < m_maxResults; ++i) {
          RemoveAction("file_result_" + std::to_string(i));
      }

      // 添加新的搜索结果
      for (size_t i = 0; i < results.size(); ++i) {
          const auto& result = results[i];
          std::string actionId = "file_result_" + std::to_string(i);
          std::string subtitle = result.directory + " (" + FormatFileSize(result.fileSize) + ")";

          AddAction(actionId, result.fileName, subtitle, "", "FileSearch", 0);
      }
  }

  double FileSearchPlugin::CalculateRelevanceScore(const std::string& fileName, const std::string& query) {
      double score = 0.0;

      // 完全匹配得分最高
      if (PluginHelper::StringContains(fileName, query, true)) {
          score += 1.0;
      }

      // 开头匹配得分较高
      if (PluginHelper::StringStartsWith(fileName, query, true)) {
          score += 0.5;
      }

      // 根据匹配位置调整得分
      std::string lowerFileName = PluginHelper::ToLowerCase(fileName);
      std::string lowerQuery = PluginHelper::ToLowerCase(query);

      size_t pos = lowerFileName.find(lowerQuery);
      if (pos != std::string::npos) {
          // 越靠前的匹配得分越高
          score += (1.0 - (double)pos / lowerFileName.length()) * 0.3;
      }

      return score;
  }

  void FileSearchPlugin::OnActionExecuted(const std::string& actionId) {
      if (PluginHelper::StringStartsWith(actionId, "file_result_")) {
          // 提取结果索引
          std::string indexStr = actionId.substr(12); // "file_result_"的长度是12
          int index = std::stoi(indexStr);

          // 从缓存中获取对应的文件路径
          if (m_searchCache.count(m_lastQuery) > 0) {
              const auto& results = m_searchCache[m_lastQuery];
              if (index >= 0 && index < results.size()) {
                  const auto& result = results[index];

                  // 打开文件或显示文件位置
                  std::string command = "explorer.exe /select,\"" + result.filePath + "\"";
                  system(command.c_str());
              }
          }
      }
  }

  void CalculatorPlugin::OnInputReceived(const std::string& input) {
      if (IsValidExpression(input)) {
          CalculationResult result = EvaluateExpression(input);

          if (result.isValid) {
              m_lastExpression = input;
              m_lastResult = result;

              // 添加计算结果动作
              std::string actionTitle = result.expression + " = " + result.formattedResult;
              AddAction("calc_result", actionTitle, "点击复制结果到剪贴板", "", "Calculator", 1);

              // 添加到历史记录
              m_history.push_back(result);
              if (m_history.size() > m_maxHistorySize) {
                  m_history.erase(m_history.begin());
              }
          } else {
              // 显示错误
              AddAction("calc_error", "计算错误: " + result.error, "", "", "Calculator", 0);
          }
      }
  }

  bool CalculatorPlugin::IsValidExpression(const std::string& input) {
      // 使用正则表达式检查是否为有效的数学表达式
      std::regex mathPattern(R"(^[\d\+\-\*/\(\)\.\s,sincostanlgqrtabef]+$)", std::regex_constants::icase);
      return std::regex_match(input, mathPattern) &&
             (input.find_first_of("0123456789") != std::string::npos);
  }

  CalculatorPlugin::CalculationResult CalculatorPlugin::EvaluateExpression(const std::string& expression) {
      CalculationResult result;
      result.expression = expression;

      try {
          // 替换常量
          std::string processedExpr = expression;
          for (const auto& constant : m_constants) {
              size_t pos = 0;
              while ((pos = processedExpr.find(constant.first, pos)) != std::string::npos) {
                  processedExpr.replace(pos, constant.first.length(), std::to_string(constant.second));
                  pos += std::to_string(constant.second).length();
              }
          }

          // 计算表达式
          result.value = ParseExpression(processedExpr);
          result.formattedResult = FormatNumber(result.value);
          result.isValid = true;

      } catch (const std::exception& ex) {
          result.error = ex.what();
          result.isValid = false;
      }

      return result;
  }

  void CalculatorPlugin::OnActionExecuted(const std::string& actionId) {
      if (actionId == "calc_result" && m_lastResult.isValid) {
          // 复制结果到剪贴板
          if (OpenClipboard(nullptr)) {
              EmptyClipboard();

              std::string result = m_lastResult.formattedResult;
              HGLOBAL hGlobal = GlobalAlloc(GMEM_MOVEABLE, result.size() + 1);
              if (hGlobal) {
                  char* pGlobal = static_cast<char*>(GlobalLock(hGlobal));
                  strcpy_s(pGlobal, result.size() + 1, result.c_str());
                  GlobalUnlock(hGlobal);
                  SetClipboardData(CF_TEXT, hGlobal);
              }

              CloseClipboard();
              ShowNotification("计算器", "结果已复制到剪贴板: " + result);
          }
      }
