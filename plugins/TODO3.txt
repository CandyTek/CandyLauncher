● 2. 简化的插件管理器 (SimplePluginManager.h)

  #pragma once
  #include "SimplePlugin.h"
  #include <map>
  #include <vector>
  #include <memory>
  #include <windows.h>

  class SimplePluginManager : public ISimplePluginHost {
  private:
      struct PluginInfo {
          HMODULE handle = nullptr;
          std::unique_ptr<ISimplePlugin> plugin;
          std::string name;
          std::string version;
          std::string filePath;
          bool loaded = false;

          CreateSimplePluginFunc createFunc = nullptr;
          DestroySimplePluginFunc destroyFunc = nullptr;
          GetSimplePluginApiVersionFunc getVersionFunc = nullptr;
      };

      std::map<std::string, PluginInfo> m_plugins;
      std::vector<SimpleActionItem> m_allActions;  // 所有插件的动作列表

      // 回调函数 - 用于通知主程序更新索引
      std::function<void()> m_onActionsChanged;

  public:
      SimplePluginManager() = default;
      ~SimplePluginManager();

      // 插件管理
      bool LoadPlugin(const std::string& dllPath);
      bool UnloadPlugin(const std::string& pluginName);
      void LoadAllPlugins(const std::string& pluginDir);
      void UnloadAllPlugins();

      // 插件操作
      void RefreshAllActions();                    // 刷新所有插件的动作
      void DispatchUserInput(const std::string& input);
      void DispatchActionExecute(const std::string& actionId);

      // 获取信息
      std::vector<std::string> GetLoadedPlugins() const;
      std::vector<SimpleActionItem> GetAllActions() const { return m_allActions; }
      std::vector<SimpleActionItem> GetPluginActions(const std::string& pluginName) const;

      // 事件回调设置
      void SetActionsChangedCallback(std::function<void()> callback) {
          m_onActionsChanged = callback;
      }

      // ISimplePluginHost 接口实现
      void AddAction(const SimpleActionItem& item) override;
      void RemoveAction(const std::string& itemId) override;
      void UpdateAction(const SimpleActionItem& item) override;
      void ClearActions(const std::string& pluginName) override;
      void ShowMessage(const std::string& title, const std::string& message) override;

  private:
      bool LoadSinglePlugin(const std::string& dllPath, PluginInfo& info);
      void UnloadSinglePlugin(PluginInfo& info);
      std::string GetPluginNameFromPath(const std::string& dllPath);
      void NotifyActionsChanged();
  };

  // SimplePluginManager.cpp
  #include "SimplePluginManager.h"
  #include <filesystem>
  #include <algorithm>

  SimplePluginManager::~SimplePluginManager() {
      UnloadAllPlugins();
  }

  bool SimplePluginManager::LoadPlugin(const std::string& dllPath) {
      std::string pluginName = GetPluginNameFromPath(dllPath);

      // 检查是否已经加载
      if (m_plugins.find(pluginName) != m_plugins.end()) {
          return false;
      }

      PluginInfo info;
      info.filePath = dllPath;
      info.name = pluginName;

      if (LoadSinglePlugin(dllPath, info)) {
          m_plugins[pluginName] = std::move(info);
          return true;
      }

      return false;
  }

  bool SimplePluginManager::LoadSinglePlugin(const std::string& dllPath, PluginInfo& info) {
      // 加载DLL
      info.handle = LoadLibraryA(dllPath.c_str());
      if (!info.handle) {
          return false;
      }

      // 获取导出函数
      info.createFunc = (CreateSimplePluginFunc)GetProcAddress(info.handle, "CreateSimplePlugin");
      info.destroyFunc = (DestroySimplePluginFunc)GetProcAddress(info.handle, "DestroySimplePlugin");
      info.getVersionFunc = (GetSimplePluginApiVersionFunc)GetProcAddress(info.handle, "GetSimplePluginApiVersion");

      if (!info.createFunc || !info.destroyFunc) {
          FreeLibrary(info.handle);
          info.handle = nullptr;
          return false;
      }

      // 创建插件实例
      ISimplePlugin* pluginPtr = info.createFunc();
      if (!pluginPtr) {
          FreeLibrary(info.handle);
          info.handle = nullptr;
          return false;
      }

      info.plugin.reset(pluginPtr);

      // 初始化插件
      if (!info.plugin->Initialize(this)) {
          info.destroyFunc(pluginPtr);
          info.plugin.reset();
          FreeLibrary(info.handle);
          info.handle = nullptr;
          return false;
      }

      info.version = info.plugin->GetVersion();
      info.loaded = true;

      // 加载插件的动作
      info.plugin->LoadActions();

      return true;
  }

  bool SimplePluginManager::UnloadPlugin(const std::string& pluginName) {
      auto it = m_plugins.find(pluginName);
      if (it == m_plugins.end()) {
          return false;
      }

      // 清除该插件的所有动作
      ClearActions(pluginName);

      // 卸载插件
      UnloadSinglePlugin(it->second);
      m_plugins.erase(it);

      NotifyActionsChanged();
      return true;
  }

  void SimplePluginManager::UnloadSinglePlugin(PluginInfo& info) {
      if (info.plugin) {
          info.plugin->Shutdown();
          if (info.destroyFunc) {
              info.destroyFunc(info.plugin.release());
          }
      }

      if (info.handle) {
          FreeLibrary(info.handle);
          info.handle = nullptr;
      }

      info.loaded = false;
  }

  void SimplePluginManager::LoadAllPlugins(const std::string& pluginDir) {
      if (!std::filesystem::exists(pluginDir)) {
          return;
      }

      for (const auto& entry : std::filesystem::directory_iterator(pluginDir)) {
          if (entry.is_regular_file() && entry.path().extension() == ".dll") {
              LoadPlugin(entry.path().string());
          }
      }

      NotifyActionsChanged();
  }

  void SimplePluginManager::UnloadAllPlugins() {
      for (auto& pair : m_plugins) {
          UnloadSinglePlugin(pair.second);
      }
      m_plugins.clear();
      m_allActions.clear();
      NotifyActionsChanged();
  }

  void SimplePluginManager::RefreshAllActions() {
      m_allActions.clear();

      for (auto& pair : m_plugins) {
          if (pair.second.loaded && pair.second.plugin) {
              pair.second.plugin->LoadActions();
          }
      }

      NotifyActionsChanged();
  }

  void SimplePluginManager::DispatchUserInput(const std::string& input) {
      for (auto& pair : m_plugins) {
          if (pair.second.loaded && pair.second.plugin) {
              pair.second.plugin->OnUserInput(input);
          }
      }
  }

  void SimplePluginManager::DispatchActionExecute(const std::string& actionId) {
      for (auto& pair : m_plugins) {
          if (pair.second.loaded && pair.second.plugin) {
              pair.second.plugin->OnActionExecute(actionId);
          }
      }
  }

  // ISimplePluginHost 接口实现
  void SimplePluginManager::AddAction(const SimpleActionItem& item) {
      // 检查是否已存在
      auto it = std::find_if(m_allActions.begin(), m_allActions.end(),
                            [&](const SimpleActionItem& action) {
                                return action.id == item.id;
                            });

      if (it != m_allActions.end()) {
          *it = item;  // 更新现有项
      } else {
          m_allActions.push_back(item);  // 添加新项
      }

      NotifyActionsChanged();
  }

  void SimplePluginManager::RemoveAction(const std::string& itemId) {
      m_allActions.erase(
          std::remove_if(m_allActions.begin(), m_allActions.end(),
                        [&](const SimpleActionItem& item) {
                            return item.id == itemId;
                        }),
          m_allActions.end());

      NotifyActionsChanged();
  }

  void SimplePluginManager::UpdateAction(const SimpleActionItem& item) {
      AddAction(item);  // AddAction 已经处理了更新逻辑
  }

  void SimplePluginManager::ClearActions(const std::string& pluginName) {
      m_allActions.erase(
          std::remove_if(m_allActions.begin(), m_allActions.end(),
                        [&](const SimpleActionItem& item) {
                            return item.id.find(pluginName + "_") == 0;
                        }),
          m_allActions.end());

      NotifyActionsChanged();
  }

  void SimplePluginManager::ShowMessage(const std::string& title, const std::string& message) {
      MessageBoxA(nullptr, message.c_str(), title.c_str(), MB_OK | MB_ICONINFORMATION);
  }

  std::vector<std::string> SimplePluginManager::GetLoadedPlugins() const {
      std::vector<std::string> result;
      for (const auto& pair : m_plugins) {
          if (pair.second.loaded) {
              result.push_back(pair.first);
          }
      }
      return result;
  }

  std::vector<SimpleActionItem> SimplePluginManager::GetPluginActions(const std::string& pluginName) const {
      std::vector<SimpleActionItem> result;
      for (const auto& action : m_allActions) {
          if (action.id.find(pluginName + "_") == 0) {
              result.push_back(action);
          }
      }
      return result;
  }

  std::string SimplePluginManager::GetPluginNameFromPath(const std::string& dllPath) {
      std::filesystem::path path(dllPath);
      return path.stem().string();
  }

  void SimplePluginManager::NotifyActionsChanged() {
      if (m_onActionsChanged) {
          m_onActionsChanged();
      }
  }

          auto pluginActions = m_pluginManager->GetAllActions();
          for (const auto& action : pluginActions) {
              // 转换为你的Action结构并添加到allActions

              pluginAction.actionName = action.title;

  #include <windows.h>

  bool ExamplePlugin::Initialize(ISimplePluginHost* host) {
      m_host = host;
      return m_host != nullptr;
  }

  void ExamplePlugin::Shutdown() {
      if (m_host) {
          m_host->ClearActions(GetName());
      }
      m_host = nullptr;
  }

  void ExamplePlugin::LoadActions() {
      if (!m_host) return;

      // 添加一些固定的动作
      SimpleActionItem action1;
      action1.id = "ExamplePlugin_hello";
      action1.title = "Hello World";
      action1.subtitle = "显示Hello World消息";
      action1.priority = 0;
      m_host->AddAction(action1);

      SimpleActionItem action2;
      action2.id = "ExamplePlugin_time";
      action2.title = "显示时间";
      action2.subtitle = "显示当前系统时间";
      action2.priority = 1;
      m_host->AddAction(action2);
  }

  void ExamplePlugin::OnUserInput(const std::string& input) {
      if (!m_host) return;

      // 根据用户输入动态添加动作
      if (input.find("test") != std::string::npos) {
          SimpleActionItem action;
          action.id = "ExamplePlugin_dynamic_test";
          action.title = "测试: " + input;
          action.subtitle = "这是动态生成的测试动作";
          action.priority = 2;
          m_host->AddAction(action);
      }

      if (input.find("calc") != std::string::npos) {
          SimpleActionItem action;
          action.id = "ExamplePlugin_calculator";
          action.title = "计算器";
          action.subtitle = "打开系统计算器";
          action.priority = 3;
          m_host->AddAction(action);
      }
  }

  void ExamplePlugin::OnActionExecute(const std::string& actionId) {
      if (!m_host) return;

      if (actionId == "ExamplePlugin_hello") {
          m_host->ShowMessage("示例插件", "Hello World from Plugin!");
      }
      else if (actionId == "ExamplePlugin_time") {
          SYSTEMTIME st;
          GetSystemTime(&st);
          char timeStr[256];
          sprintf_s(timeStr, "当前时间: %04d-%02d-%02d %02d:%02d:%02d UTC",
                   st.wYear, st.wMonth, st.wDay, st.wHour, st.wMinute, st.wSecond);
          m_host->ShowMessage("时间", timeStr);
      }
      else if (actionId == "ExamplePlugin_dynamic_test") {
          m_host->ShowMessage("测试", "这是动态生成的测试动作被执行了！");
      }
      else if (actionId == "ExamplePlugin_calculator") {
          // 打开系统计算器

  API说明

  SimpleActionItem 结构

  - id: 唯一标识符（建议格式：插件名_动作名）
  - title: 显示标题
  - subtitle: 副标题/描述
  - iconPath: 图标路径（可选）
  - priority: 优先级（数字越小优先级越高）

  ISimplePluginHost 接口

  - AddAction(): 添加动作项
  - RemoveAction(): 移除动作项
  - UpdateAction(): 更新动作项
  - ClearActions(): 清除插件的所有动作
  - ShowMessage(): 显示消息对话框

  回调函数

  - LoadActions(): 插件启动时加载固定动作
  - OnUserInput(): 用户输入时调用，可动态添加动作
  - OnActionExecute(): 用户执行动作时调用

  注意事项

  1. 动作ID必须唯一，建议使用"插件名_动作名"格式
  2. 在Shutdown()中清理资源
  3. 使用GetName()来标识你的插件
  4. 导出函数名称必须准确匹配

  4. 简单接口：最小化的API，易于理解和实现

  ✅ 安全保障

  1. 接口隔离：插件不能直接访问你的allActions，只能通过接口操作
  2. 类型安全：使用强类型接口，减少错误
  3. 生命周期管理：自动管理插件的加载和卸载

  ✅ 易于集成

  1. 最小侵入：只需要在几个关键点调用插件系统
  2. 向后兼容：不影响现有代码结构
  3. 渐进增强：可以在此基础上逐步添加更多功能

  🔧 使用步骤

  1. 集成到主程序：
    - 在ListedRunnerPlugin中添加SimplePluginManager
    - 在程序启动时调用InitializePlugins()
    - 在索引刷新时调用RefreshActionsList()
    - 在用户输入时调用OnUserInput()
    - 在动作执行时调用OnActionExecute()
  2. 开发插件：
    - 继承ISimplePlugin接口
    - 实现必要的虚函数
    - 导出三个C函数
    - 编译为DLL并放置到plugins文件夹
  3. 部署使用：
    - 将插件DLL放入plugins文件夹
    - 重启程序或调用插件刷新

  这个简化版本专注于核心功能，去掉了复杂的安全机制、热加载、配置系统等高级功能，但保留了完整的DLL加载和索引集成能力。你可以先实现这个版本，验证基本功能后再逐步添加其他特性。
